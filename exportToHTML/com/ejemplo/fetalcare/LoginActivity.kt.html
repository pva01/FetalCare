18   class LoginActivity : AppCompatActivity() {
19       private lateinit var binding: ActivityLoginBinding
20       private lateinit var firebaseAuth: FirebaseAuth
21       private lateinit var authStateListener: FirebaseAuth.AuthStateListener
22   
23       override fun onCreate(savedInstanceState: Bundle?) {
24           super.onCreate(savedInstanceState)
25           binding = ActivityLoginBinding.inflate(layoutInflater)
26           setContentView(binding.root)
27   
28           firebaseAuth = FirebaseAuth.getInstance()
29   
30           authStateListener = FirebaseAuth.AuthStateListener { firebaseAuth ->
31               if (firebaseAuth.currentUser != null) {
32                   val intent = Intent(this, MainActivity::class.java)
33                   startActivity(intent)
34                   finish()
35               }
36           }
37   
38           binding.loginButton.setOnClickListener {
39               val email = binding.loginEmail.text.toString()
40               val password = binding.loginPassword.text.toString()
41   
42               if (email.isNotEmpty() && password.isNotEmpty()) {
43                   firebaseAuth.signInWithEmailAndPassword(email, password).addOnCompleteListener {
44                       if (it.isSuccessful) {
45                           Toast.makeText(this, "Inicio de sesi√≥n exitoso", Toast.LENGTH_SHORT)
46                               .show()
47                       } else {
48                           Toast.makeText(this, it.exception.toString(), Toast.LENGTH_SHORT).show()
49                       }
50                   }
51               } else {
52                   Toast.makeText(this, "Los espacios no pueden estar en blanco", Toast.LENGTH_SHORT)
53                       .show()
54               }
55           }
56   
57           binding.forgotPassword.setOnClickListener {
58               val builder = AlertDialog.Builder(this)
59               val view = layoutInflater.inflate(R.layout.dialog_forgot, null)
60               val userEmail = view.findViewById<EditText>(R.id.editBox)
61   
62               builder.setView(view)
63               val dialog = builder.create()
64   
65               view.findViewById<Button>(R.id.btnResetl).setOnClickListener {
66                   compareEmail(userEmail)
67                   dialog.dismiss()
68               }
69               view.findViewById<Button>(R.id.btnCancel).setOnClickListener {
70                   dialog.dismiss()
71               }
72               if (dialog.window != null) {
73                   dialog.window!!.setBackgroundDrawable(ColorDrawable(0))
74               }
75               dialog.show()
76           }
77   
78           binding.signupRedirectText.setOnClickListener {
79               val signupIntent = Intent(this, SignupActivity::class.java)
80               startActivity(signupIntent)
81           }
82   
83       }
84   
85       override fun onStart() {
86           super.onStart()
87           // Agregamos el AuthStateListener al inicio de la actividad
88           firebaseAuth.addAuthStateListener(authStateListener)
89       }
90   
91       override fun onStop() {
92           super.onStop()
93           // Eliminamos el AuthStateListener al detener la actividad para evitar fugas de memoria
94           firebaseAuth.removeAuthStateListener(authStateListener)
95       }
96   
97       private fun compareEmail(email: EditText) {
98           if (email.text.toString().isEmpty()) {
99               Toast.makeText(this, "El correo no puede estar vac√≠o", Toast.LENGTH_LONG).show()
100              return
101          }
102          if (!Patterns.EMAIL_ADDRESS.matcher(email.text.toString()).matches()) {
103              Toast.makeText(this, "Correo inv√°lido", Toast.LENGTH_LONG).show()
104          } else {
105              firebaseAuth.fetchSignInMethodsForEmail(email.text.toString())
106                  .addOnCompleteListener { task ->
107                      if (task.isSuccessful) {
108                          val signInMethods = task.result?.signInMethods
109                          if (signInMethods == null || signInMethods.isEmpty()) {
110                              Toast.makeText(this, "Correo no registrado", Toast.LENGTH_LONG)
111                                  .show()
112                          } else {
113                              firebaseAuth.sendPasswordResetEmail(email.text.toString())
114                                  .addOnCompleteListener { task ->
115                                      if (task.isSuccessful) {
116                                          Toast.makeText(
117                                              this,
118                                              "Revisa tu correo",
119                                              Toast.LENGTH_LONG
120                                          ).show()
121                                      }
122                                  }
123                          }
124                      }
125                  }
126          }
127      }
